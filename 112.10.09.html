// 使用 D3.js 加载 JSON 数据 d3.json("graph_data.json").then(function(data) { // 创建 SVG 元素 var svg = d3.select("#graph-container") .append("svg") .attr("width", 800) .attr("height", 800); // 为每个作者创建独特的颜色 var colorScale = d3.scaleOrdinal(d3.schemeCategory10); // 创建力导向图 var simulation = d3.forceSimulation(data.nodes) .force("link", d3.forceLink(data.links).id(function(d) { return d.id; })) .force("charge", d3.forceManyBody()) .force("center", d3.forceCenter(400, 400)); // 添加边 var link = svg.selectAll("line") .data(data.links) .enter().append("line") .style("stroke", "gray"); // 设置边的默认颜色 // 添加节点 var node = svg.selectAll("circle") .data(data.nodes) .enter().append("circle") .attr("r", function(d) { return 5 + d.degree_centrality * 20; }) .style("fill", function(d) { return colorScale(d.id); }); // 设置节点的颜色 // 添加节点标签 var label = svg.selectAll(null) .data(data.nodes) .enter().append("text") .text(function(d) { return d.id; }) .attr("font-size", 8) .attr("dx", 12) .attr("dy", 4); // 设置节点和边的位置更新函数 simulation.on("tick", function() { link.attr("x1", function(d) { return d.source.x; }) .attr("y1", function(d) { return d.source.y; }) .attr("x2", function(d) { return d.target.x; }) .attr("y2", function(d) { return d.target.y; }); node.attr("cx", function(d) { return d.x; }) .attr("cy", function(d) { return d.y; }); label.attr("x", function(d) { return d.x; }) .attr("y", function(d) { return d.y; }); }); });
